Unity сложнее веб-разработки, но вот самый простой способ проверить:

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

ШАГ 1: Установи NativeWebSocket (ОБЯЗАТЕЛЬНО!)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

В Unity:
1. Window → Package Manager
2. "+" (плюс в левом верхнем углу)
3. Add package from git URL
4. Вставь: https://github.com/endel/NativeWebSocket.git#upm
5. Нажми Add
6. Подожди установки (10-30 сек)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

ШАГ 2: Создай тестовую сцену
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. File → New Scene
2. Сохрани как "TestScene" (Ctrl+S)
3. В Hierarchy (слева):
   - Правый клик → Create Empty
   - Назови "TestObject"
4. Выбери TestObject
5. В Inspector (справа):
   - Add Component
   - Найди "SimpleTest"
   - Добавь его

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

ШАГ 3: Запусти тест
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Убедись что Go сервер запущен (START_GAME.bat)
2. Нажми Play ▶️ в Unity
3. Открой Console (Window → General → Console)
4. Смотри логи:

   ✅ Если видишь "CONNECTION SUCCESSFUL!" - всё работает!
   ❌ Если "CONNECTION ERROR" - сервер не запущен

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

ПОЧЕМУ Unity СЛОЖНЕЕ ЧЕМ WEB?
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Web/Mobile:
  ✅ npm install → npm start → работает
  ✅ Код сразу выполняется
  ✅ Простая отладка в браузере

Unity:
  ⚠️ Нужно устанавливать пакеты вручную
  ⚠️ Код не работает без GameObject в сцене
  ⚠️ Нужно создавать сцены, префабы, assets
  ⚠️ Компиляция + импорт assets = долго

НО! После первой настройки всё становится проще.

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

TROUBLESHOOTING
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Ошибка: "NativeWebSocket not found"
→ Установи пакет через Package Manager (Шаг 1)

Ошибка: "CONNECTION ERROR"
→ Запусти START_GAME.bat (сервер должен работать)

Ошибка: "SimpleTest not found"
→ Подожди пока Unity скомпилирует скрипт (5-10 сек)

Нет логов в Console
→ Открой Window → General → Console

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
